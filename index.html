🔧 后端准备清单（需在 Supabase Dashboard 配置）

-- 1. 给 posts 表添加 cover_url 和 summary 字段
ALTER TABLE posts ADD COLUMN IF NOT EXISTS cover_url text;
ALTER TABLE posts ADD COLUMN IF NOT EXISTS summary text;

-- 2. 创建 comments 表
CREATE TABLE IF NOT EXISTS comments (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  post_id uuid REFERENCES posts(id) ON DELETE CASCADE,
  user_id uuid REFERENCES auth.users(id),
  content text NOT NULL,
  created_at timestamptz DEFAULT now()
);

-- 3. 启用 RLS
ALTER TABLE comments ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Anyone can read comments" ON comments FOR SELECT USING (true);
CREATE POLICY "Auth users can insert comments" ON comments FOR INSERT WITH CHECK (auth.uid() = user_id);

-- 4. 创建 Storage bucket（在 Storage 面板操作）
-- 名称: post-covers, Public bucket: true

-- 5. 启用 pg_graphql 扩展
CREATE EXTENSION IF NOT EXISTS pg_graphql;

-- 6. 部署 Edge Function (supabase functions deploy generate-summary)
-- 参见项目 supabase/functions/generate-summary/index.ts
